    <script src="https://cdn.jsdelivr.net/gh/studio-freight/lenis@1.0.29/bundled/lenis.min.js"></script>

    <script>
        // Lenis Smooth Scroll
        const lenis = new Lenis({
            duration: 1.2,
            easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
            direction: 'vertical',
            gestureDirection: 'vertical',
            smooth: true,
            mouseMultiplier: 1,
            smoothTouch: false,
        });

        function raf(time) {
            lenis.raf(time);
            requestAnimationFrame(raf);
        }
        requestAnimationFrame(raf);

        // Register GSAP Plugins
        gsap.registerPlugin(ScrollTrigger);

        // 1. HERO ANIMATION
        const heroTl = gsap.timeline();
        heroTl.from(".reveal-heading", { y: 50, opacity: 0, duration: 1, stagger: 0.2, ease: "power3.out" })
              .from(".reveal-text", { y: 30, opacity: 0, duration: 0.8 }, "-=0.8")
              .from(".reveal-buttons", { y: 30, opacity: 0, duration: 0.8 }, "-=0.6")
              .from(".hero-card-anim", { scale: 0.8, opacity: 0, duration: 0.8 }, "-=0.6");

        // 2. PARALLAX IMAGES
        // Effect: Image moves slower than the scroll (yPercent)
        gsap.to(".parallax-img", {
            yPercent: 20,
            ease: "none",
            scrollTrigger: {
                trigger: "#hero",
                start: "top top",
                end: "bottom top",
                scrub: true
            }
        });

        gsap.utils.toArray(".parallax-img-team").forEach(img => {
            gsap.to(img, {
                yPercent: 15,
                ease: "none",
                scrollTrigger: {
                    trigger: img,
                    start: "top bottom",
                    end: "bottom top",
                    scrub: true
                }
            });
        });

        // 3. CHALLENGE CARDS REVEAL
        gsap.from(".challenge-col", {
            scrollTrigger: { trigger: ".challenge-col", start: "top 85%" },
            y: 50,
            opacity: 0,
            duration: 0.8,
            stagger: 0.2
        });

        // 4. HORIZONTAL SCROLL (CONSULTING SECTION) - DESKTOP ONLY
        if (window.innerWidth > 991) {
            let panels = gsap.utils.toArray(".service-panel");
            
            gsap.to(panels, {
                xPercent: -100 * (panels.length - 1),
                ease: "none",
                scrollTrigger: {
                    trigger: "#consulting-horizontal-wrapper",
                    pin: true,
                    scrub: 1,
                    end: () => "+=" + document.querySelector(".scrolling-content").offsetWidth
                }
            });
        }

        // 5. TIMELINE LINE DRAW
        const timelineItems = document.querySelectorAll('.timeline-item');
        timelineItems.forEach((item, index) => {
            ScrollTrigger.create({
                trigger: item,
                start: "top 80%",
                onEnter: () => {
                    item.classList.add('active');
                    // Draw the line
                    const progress = ((index + 1) / timelineItems.length) * 100;
                    gsap.to('.timeline-progress', { height: `${progress}%`, duration: 0.5 });
                }
            });
        });

        // 6. PLATFORM PIN & BADGES
        ScrollTrigger.create({
            trigger: ".dashboard-visual",
            start: "top 15%",
            endTrigger: "#platform",
            end: "bottom bottom",
            pin: true,
            pinSpacing: false
        });

        document.querySelectorAll('.feature-trigger').forEach((trigger) => {
            const targetId = trigger.getAttribute('data-target');
            const targetEl = document.getElementById(targetId);
            
            ScrollTrigger.create({
                trigger: trigger,
                start: "top 60%",
                end: "bottom 60%",
                onEnter: () => gsap.to(targetEl, { opacity: 1, scale: 1, duration: 0.5 }),
                onLeave: () => gsap.to(targetEl, { opacity: 0, scale: 0.8, duration: 0.3 }),
                onEnterBack: () => gsap.to(targetEl, { opacity: 1, scale: 1, duration: 0.5 }),
                onLeaveBack: () => gsap.to(targetEl, { opacity: 0, scale: 0.8, duration: 0.3 })
            });
        });

        // 7. SPLIT MERGE
        const mergeTl = gsap.timeline({
            scrollTrigger: { trigger: "#merge", start: "top top", end: "bottom bottom", scrub: 1 }
        });

        mergeTl.to(".split-left", { width: "0%", opacity: 0, padding: 0 }, "merge")
               .to(".split-right", { width: "100%", duration: 1 }, "merge")
               .to(".merge-icon-center", { opacity: 1, scale: 1.2, duration: 0.5 }, "merge+=0.5")
               .to(".merge-result-text", { opacity: 1, y: -50, duration: 1 }, "merge+=0.5");

        // 8. COUNTERS
        gsap.utils.toArray('.counter').forEach(counter => {
            const target = counter.getAttribute('data-target');
            gsap.to(counter, {
                innerText: target,
                duration: 2,
                snap: { innerText: 1 },
                scrollTrigger: { trigger: counter, start: "top 85%", once: true }
            });
        });

        // 9. GENERAL HEADING REVEAL
        gsap.utils.toArray(".reveal-heading").forEach(heading => {
            gsap.from(heading, {
                scrollTrigger: { trigger: heading, start: "top 90%" },
                y: 30,
                opacity: 0,
                duration: 0.8
            });
        });

    </script>
